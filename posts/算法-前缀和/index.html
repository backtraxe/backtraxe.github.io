static int[] getPrefixSum(int[] nums) {
    int n = nums.length;
    int[] pre = new int[n + 1];
    for (int i = 0; i < n; i++) pre[i + 1] = pre[i] + nums[i];
    return pre;
}

class Solution {
    public int subarraySum(int[] nums, int k) {
        int n = nums.length;
        int[] pre = new int[n + 1];
        for (int i = 0; i < n; i++) pre[i + 1] = pre[i] + nums[i];
        HashMap<Integer, Integer> map = new HashMap<>();
        int ans = 0;
        for (int x : pre) {
            ans += map.getOrDefault(x - k, 0); // pre[j + 1] - pre[i] == k
            map.put(x, map.getOrDefault(x, 0) + 1);
        }
        return ans;
    }
}

class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        Arrays.fill(ans, 1);
        int left = 1;
        int right = 1;
        for (int i = 0; i < n; i++) {
            ans[i] *= left;
            left *= nums[i];
            ans[n - 1 - i] *= right;
            right *= nums[n - 1 - i];
        }
        return ans;
    }
}
