class Solution {
    public int findLongestChain(int[][] pairs) {
        int n = pairs.length;
        int[] dp = new int[n];
        Arrays.sort(pairs, (a, b) -> a[0] - b[0]);
        for (int i = 0; i < n; i++) {
            int maxLen = 0;
            for (int j = 0; j < i; j++)
                if (pairs[j][1] < pairs[i][0])
                    maxLen = Math.max(maxLen, dp[j]);
            dp[i] = maxLen + 1;
        }
        return dp[n - 1];
    }
}

class Solution {
    public int findLongestChain(int[][] pairs) {
        int n = pairs.length;
        int[] dp = new int[n];
        Arrays.fill(dp, Integer.MAX_VALUE);
        int ans = 0;
        Arrays.sort(pairs, (a, b) -> a[0] - b[0]);
        for (int[] p : pairs) {
            // æŸ¥æ‰¾ç¬¬ä¸€ä¸ªå¤§äºç­‰äº p[0] çš„ä¸‹æ ‡
            int l = 0;
            int r = n;
            while (l < r) {
                int m = l + (r - l) / 2;
                if (dp[m] >= p[0]) r = m;
                else l = m + 1;
            }
            dp[l] = Math.min(dp[l], p[1]);
            ans = Math.max(ans, l + 1);
        }
        return ans;
    }
}

class Solution {
    public int findLongestChain(int[][] pairs) {
        Arrays.sort(pairs, (a, b) -> a[0] - b[0]);
        int[] pre = pairs[0];
        int ans = 1;
        for (int[] p : pairs) {
            if (pre[1] < p[0] || pre[1] > p[1]) {
                if (pre[1] < p[0]) ans++;
                pre = p;
            }
        }
        return ans;
    }
}
