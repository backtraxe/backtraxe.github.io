// 数据结构定义
// struct timeval {
//     long tv_sec;  // 秒
//     long tv_usec; // 微秒（百万分之一秒）
// };

struct timeval st_time, ed_time;
gettimeofday(&st_time, NULL);
// 其他程序
gettimeofday(&ed_time, NULL);

// 以秒为单位
double time_dif = ed_time.tv_sec - st_time.tv_sec + (ed_time.tv_usec - st_time.tv_usec) / 1000000.0;
