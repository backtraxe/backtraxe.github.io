class Solution {
    public int preimageSizeFZF(int k) {
        int l = 0;
        int r = k;
        while (l <= r) {
            int m = l + (r - l) / 2;
            long x = m * 5L;
            int cnt = 0;
            while (x > 0) {
                cnt += x / 5;
                x /= 5;
            }
            if (cnt == k) return 5;
            else if (cnt > k) r = m - 1;
            else l = m + 1;
        }
        return 0;
    }
}
